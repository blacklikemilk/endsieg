#White peace, information
country_event = {
	id = endsieg_achieved.1
	title = endsieg_achieved.1.t
	picture = GFX_report_event_german_troops
	desc = {
		text = endsieg_achieved.1.difficultyeasy
		trigger = {
			has_idea = easy_mode
		}
	}
	desc = {
		text = endsieg_achieved.1.difficultynormal
		trigger = {
			NOT = {
				has_idea = easy_mode
				has_idea = hard_mode
				has_idea = extreme_hard_mode
			}
		}
	}
	desc = {
		text = endsieg_achieved.1.difficultyhard
		trigger = {
			has_idea = hard_mode
		}
	}
	desc = {
		text = endsieg_achieved.1.difficultyextremehard
		trigger = {
			has_idea = extreme_hard_mode
		}
	}

	fire_only_once = yes

	trigger = {
		tag = GER
		GER = { has_war_with = SOV }
		has_global_flag = welcome_event_WW2
		NOT = { has_global_flag = player_wants_custom_peacedeals }
		NOT = {
			SOV = {
				controls_state = 137
				controls_state = 195
				controls_state = 217
				controls_state = 219
				controls_state = 760
			}
		}
		SOV = { surrender_progress > 0.85 }
	}

	option = {
		name = endsieg_achieved.1.a
		every_country = {
			limit = { has_war_with = SOV }
			white_peace = SOV
		}
		hidden_effect = {
			country_event = { id = endsieg_achieved.3 days = 1}
		}
	}

	option = {
		name = endsieg_achieved.1.b
	}
}
