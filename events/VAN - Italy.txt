					OR = {
						is_major = yes
						tag = YUG
					}
					NOT = { tag = ITA }
					OR = {
						has_guaranteed = ALB
						is_in_faction_with = ALB
					}
				}
			}
			modifier = {
				factor = 0.5
				OR = {
					has_government_communist = yes
					has_government_nationalist = yes #glorious Albania will never yield!
				}
			}
		}
		ITA = {
			country_event = { days = 1 id = vanilla_italy.1 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
	option = { # Resist
		name = vanilla_italy.3.b
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = ITA
						ITA = { is_faction_leader = yes }
					}
					is_historical_focus_on = yes
					AND = {
						is_in_faction = no
						NOT = {
							any_other_country = {
								NOT = { tag = ITA }
								has_guaranteed = ALB
							}
						}
					}
				}
			}
		}
		ITA = {
			effect_tooltip = {
				create_wargoal = {
					type = take_state_focus
					target = ALB
					generator = { 44 }
				}
			}
			country_event = { days = 1 id = vanilla_italy.2 }
		}
	}
}

		ai_chance = {
			factor = 33
			modifier = {
				factor = 3
				has_army_size = { size > 75 }
			}
		}
		name = vanilla_italy.12.a
		ITA = {
			add_political_power = -100
			if = { limit = { is_faction_leader = yes }
				add_to_faction = YUG

					add_ai_strategy = {
						type = alliance
						id = "ITA"
						value = 200
					}
				}
			}
			if = {
				limit = {
					OR = {
						AND = {
							is_faction_leader = no
							is_in_faction = yes
						}
						is_in_faction = no
					}
				}
				create_faction = vanilla_italy_empire_faction
				add_to_faction = YUG
				YUG = {
					add_ai_strategy = {
						type = alliance
						id = "ITA"
						value = 200
					}
				}
			}
		}
		GER = { country_event = { id = germany.96 hours = 6 } }
	}
	option = {
		ai_chance = {
			factor = 100
		}
		name = vanilla_italy.12.b
		YUG = {
			add_opinion_modifier = { target = ITA modifier = western_betrayal }

	}
