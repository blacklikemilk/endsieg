#Sinking of Brazillian Merchant fleet!
country_event = {
	id = ww1_brazil.4
	title = ww1_brazil.4.t
	desc = ww1_brazil.4.d
	#picture =
	fire_only_once = yes

	trigger = {
		has_global_flag = unrestricted_submarine_warfare
		NOT = {has_global_flag = GE1_backs_down}
		original_tag = BR1
		GE1 = {
			OR = {
				has_war_with = FR3
				has_war_with = EN1
			}
		}
	}

	mean_time_to_happen = { days = 40 }

	immediate = {
		add_equipment_to_stockpile = {
			type = convoy_1
			amount = -5
			producer = BR1
		}
	}
	option = {
		name = ww1_brazil.4.a
		ai_chance = {factor = 100}
		BR1 = {add_to_faction = entente}
		BR1 = {
			declare_war_on = {
				target = GE1
				type = annex_everything
			}
			add_stability = 0.40
			add_war_support = 0.20
		}
	}
	option = {
		name = ww1_brazil.4.b
		ai_chance = {factor = 0}
		add_stability = -0.10
		add_war_support =  0.05
	}
}
