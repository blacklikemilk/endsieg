# Stalinism on the Rise
country_event = {
	id = endsieg_political.1
	title = endsieg_political.1.t
	desc = endsieg_political.1.d
	picture = GFX_report_event_worker_protests
	
	is_triggered_only = yes

	trigger = {
		NOT = { has_government = stalinist }
	}
	
	# Seek government support
	option = {
		name = endsieg_political.1.a
		trigger = {
			OR = {
				NOT = { tag = CHI }
				AND = {
					NOT = { country_exists = PRC }
					tag = CHI
				}			
			}
		}
		custom_effect_tooltip = endsieg_political.1.a.tt
		clr_country_flag = communism_popular_movement
		set_country_flag = communism_endsieg_political.movement
	}

	# Seek popular support
	option = {
		name = endsieg_political.1.b
		custom_effect_tooltip = endsieg_political.1.b.tt
		clr_country_flag = communism_endsieg_political.movement
		clr_country_flag = communism_government_support
		set_country_flag = communism_popular_movement
	}
}

# Communists Execute Coup
country_event = {
	id = endsieg_political.3
	title = endsieg_political.3.t
	desc = endsieg_political.3.d
	picture = GFX_report_event_gathering_protest
	
	trigger = {
		has_idea_with_trait = communism_revolutionary
		NOT = { has_government = stalinist }
		has_country_flag = communism_government_support
		stalinist > 0.4
		is_major = no
		is_puppet = no
		OR = {
			NOT = { tag = CHI }
			AND = {
				NOT = { country_exists = PRC }
				tag = CHI
			}			
		}
	}

	mean_time_to_happen = {
		days = 100
		modifier = {
			factor = 0.5
			stalinist > 0.5
		}
	}

	immediate = {
		hidden_effect = { remove_ideas = communist_revolutionaries }
	}
	
	option = {
		name = endsieg_political.3.a
		set_politics = {
			ruling_party = stalinist
			elections_allowed = no
		}
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
}
