add_namespace = romania

# Treaty of Bucharest
country_event = {
	id = romania.1
	title = romania.1.t
	desc = romania.1.d
	picture = GFX_report_event_treaty_of_bucharest

	fire_only_once = yes

	trigger = {
		date < 1919.1.1
		tag = RMM
		date > 1918.5.7
		country_exists = BNN
		NOT = { has_global_flag = bucharestreaty }
		NOT = { has_global_flag = welcome_event_1910 }
		NOT = { has_global_flag = welcome_event_1914 }
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = romania.1.a
		news_event = { hours = 6 id = news.5 }
		set_global_flag = bucharestreaty
		BNN = {
			transfer_state = 77 #Basically Dobrudja
			transfer_state = 759
			transfer_state = 778
		}
		AUH = {
			transfer_state = 748 #Carpathian passages
		}
	}
}

# Union of Romania with Bessarabia
country_event = {
	id = romania.3
	title = romania.3.t
	desc = romania.3.d
	picture = GFX_report_event_bessarabia

	fire_only_once = yes

	trigger = {
		date < 1919.1.1
		tag = RMM
		date > 1918.4.9
		NOT = { has_global_flag = bessarabia }
		RO2 = { is_puppet_of = RMM }
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = romania.3.a
		news_event = { hours = 6 id = news.18 }
		RMM = {
			annex_country = { target = RO2 transfer_troops = yes }
		}
	}
}
